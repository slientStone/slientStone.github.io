echarts图表使用-折线图的属性

2020-07-29

​
折现图中，相关参数记录

  title-text：显示标题，title-left:center：可居中文字，grid-top：调整上高度，tooltip-formatter：重设鼠标指示显示文字。

 为实现以下类似功能，需要设置tooltip-formatter：

this.chart.setOption({
    title: {
      text: '记录'
      // left: 'center',
      //         align: 'right'
    },
    xAxis: {
      data: xData,
      boundaryGap: false,
      type: 'category'
    },
    grid: {
      left: 10,
      right: 10,
      bottom: 20,
      top: 30,
      containLabel: true
    },
    tooltip: {
      trigger: 'axis',
      axisPointer: {
        type: 'cross'
      },
      padding: [5, 10],
      // formatter: tooltipsFormatter
      formatter: function(params, ticket, dataIndex) {
        let res = params[0].name;
        for (let i = 0, l = params.length; i < l; i++) {
          res += '<br/>' + params[i].marker + params[i].seriesName + ' : ' + 
          params[i].value + '/' + params[i].data.money;
        }
        return res;
      }
    },
    yAxis: {
      axisTick: {
        show: false
      },
      axisLabel: {
        formatter: '{value}'
      }
    },
    legend: {
      data: legendData,
      formatter: '{name}'
    },
    series: seriesData
  });
